% Bold title left + italic date right; omit date if empty
\newcommand{\entryheading}[2]{%
  \noindent\textbf{#1}\ifx&#2&\else\hfill\textit{#2}\fi\par
}

% Italic institution left + detail right, with spacing below
\newcommand{\entrysubheading}[2]{%
  \noindent\textit{#1}\hfill#2\par\vspace{4pt}
}

% Bold label + colon + value for skills rows
\newcommand{\skill}[2]{%
  \noindent\textbf{#1:} #2\par\vspace{2pt}
}

  \end{itemize}
}
